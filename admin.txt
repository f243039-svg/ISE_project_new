    <script>
        // Init
        const currentUser = utils.checkAuth('admin');

        // Views
        function renderDashboard() {
            document.getElementById('content-area').innerHTML = `
                <div class="fade-in">
                    <div class="mb-8"><h2 class="text-2xl font-bold text-slate-800">Admin Dashboard</h2></div>
                    <div class="grid grid-cols-1 md:grid-cols-3 gap-6">
                        <div class="stat-card">
                            <h3 class="text-xs font-bold text-slate-400 uppercase tracking-wider">Total Users</h3>
                            <p class="text-3xl font-bold text-slate-800 mt-2">${db.db.users.length}</p>
                        </div>
                        <div class="stat-card">
                            <h3 class="text-xs font-bold text-slate-400 uppercase tracking-wider">Courses</h3>
                            <p class="text-3xl font-bold text-slate-800 mt-2">${db.db.courses.length}</p>
                        </div>
                    </div>
                </div>
            `;
        }

        function renderRegister() {
            document.getElementById('content-area').innerHTML = `
                <div class="fade-in">
                    <div class="mb-8"><h2 class="text-2xl font-bold text-slate-800">User Registration</h2></div>
                    <div class="stat-card max-w-2xl">
                        <form id="reg-form" class="space-y-6">
                            <div class="grid grid-cols-2 gap-6">
                                <div><label>Full Name</label><input type="text" id="r-name" required></div>
                                <div><label>User ID</label><input type="text" id="r-id" required></div>
                            </div>
                            <div class="grid grid-cols-2 gap-6">
                                <div><label>Role</label><select id="r-role"><option value="student">Student</option><option value="teacher">Teacher</option><option value="admin">Admin</option></select></div>
                                <div><label>Password</label><input type="text" id="r-pass" value="123" required></div>
                            </div>
                            <div class="pt-4"><button type="submit" class="bg-blue-900 text-white px-8 py-2.5 rounded hover:bg-blue-800 font-medium">Create Account</button></div>
                        </form>
                    </div>
                </div>
            `;
            document.getElementById('reg-form').onsubmit = (e) => {
                e.preventDefault();
                if(db.addUser({
                    id: document.getElementById('r-id').value,
                    name: document.getElementById('r-name').value,
                    role: document.getElementById('r-role').value,
                    password: document.getElementById('r-pass').value
                })) { utils.toast('User Created'); e.target.reset(); } 
                else { alert('User ID exists'); }
            };
        }

        function renderCourses() {
            const rows = db.getCourses().map(c => `<tr class="hover:bg-slate-50"><td class="font-medium">${c.code}</td><td>${c.name}</td><td>${c.credits}</td><td>${c.teacherId}</td></tr>`).join('');
            const teachers = db.getUsersByRole('teacher').map(t => `<option value="${t.id}">${t.name}</option>`).join('');
            
            document.getElementById('content-area').innerHTML = `
                <div class="fade-in">
                    <div class="mb-8"><h2 class="text-2xl font-bold text-slate-800">Course Registry</h2></div>
                    <div class="stat-card mb-8 max-w-4xl">
                        <h3 class="font-bold text-slate-700 mb-4 pb-2 border-b">Add Course</h3>
                        <form id="course-form" class="grid grid-cols-4 gap-4 items-end">
                            <div><label>Code</label><input type="text" id="c-code" required></div>
                            <div><label>Name</label><input type="text" id="c-name" required></div>
                            <div><label>Credits</label><input type="number" id="c-cr" value="3" min="1" max="4"></div>
                            <div><label>Teacher</label><select id="c-tid">${teachers}</select></div>
                            <button class="bg-blue-900 text-white px-4 py-2.5 rounded col-span-4 mt-2">Add Course</button>
                        </form>
                    </div>
                    <div class="bg-white rounded shadow-sm border overflow-hidden">
                        <table><thead><tr><th>Code</th><th>Name</th><th>Credits</th><th>Instructor</th></tr></thead><tbody>${rows}</tbody></table>
                    </div>
                </div>
            `;
            document.getElementById('course-form').onsubmit = (e) => {
                e.preventDefault();
                if(db.addCourse({
                    code: document.getElementById('c-code').value,
                    name: document.getElementById('c-name').value,
                    credits: parseInt(document.getElementById('c-cr').value),
                    teacherId: document.getElementById('c-tid').value
                })) { utils.toast('Course Added'); renderCourses(); }
                else { alert('Course code exists'); }
            };
        }

        // Load Default
        renderDashboard();
    </script>
